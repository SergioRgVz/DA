void testalgorithm()
{
    std::vector<int> v = {1, 2, 3, 4, 5, 6, 7, 8, 9};
    std::vector<int> w = {1, 2, 3, 4, 5, 6, 7, 8, 9};

    do{
        merge_sortint(v, 0, 8);
        if(!std::equal(v.begin(), v.end(), w.begin()))
        {
            std::cout << "v != w" << std::endl;
        }
    }while(std::next_permutation(v.begin(), v.end()));

    do{
        quicksortint(v, 0, 8);
        if(!std::equal(v.begin(), v.end(), w.begin()))
        {
            std::cout << "v != w" << std::endl;
        }
    }while(std::next_permutation(v.begin(), v.end()));    
}
